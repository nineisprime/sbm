\begin{lemma}
\label{lem:level_set_control}
Let $h \,:\, [0,1] \rightarrow \R$ be a measurable function. Let $t > 1$, suppose $\int (x(1-x))^{t-1} |h(x)| dx \leq M'$, then, we have that 
$\mu \{ x \,:\, |h(x)| \geq \kappa \} \leq \frac{M'+4}{\kappa^{1/t}} $ for any $\kappa > 4$. 
\end{lemma}

\begin{proof}

First, observe that, by the definition of Lebesgue integral, 
\[
\kappa \mu \{ x \,:\, (x(1-x))^{t-1} |h(x)| \geq \kappa \} \leq \int  (x(1-x))^{t-1} |h(x)| dx \leq M'
\]

Let $a \in (0,1/2)$, and suppose $1 - a \geq x \geq a$, then $x(1-x) \geq a(1-a) \geq \frac{a}{2}$. Therefore, $(x(1-x))^{t-1} \geq \left( \frac{a}{2} \right)^{t-1}$. 

\begin{align*}
\{ x \,:\, (x(1-x))^{t-1} |h(x)| \geq \kappa \} &\supset 
    \{ x \,:\, |h(x)| \geq \frac{2^{t-1} \kappa}{a^{t-1}} ,\, 1 - a \geq x \geq a \} \quad \trm{for any $a \in (0, 1/2)$} \\
  &\supset \left \{ x \,:\, |h(x)| \geq \kappa^t,\, 1 - \frac{2}{\kappa} \geq x \geq \frac{2}{\kappa} \right \} \\
 &= \{ x \,:\, |h(x)| \geq \kappa^t \} \cap \left \{ x \,:\, 1 - \frac{2}{\kappa} \geq x \geq \frac{2}{\kappa} \right \} 
\end{align*}


Therefore, we have that
\begin{align*}
\mu \{ x \,:\, (x(1-x))^{t-1} |h(x)| \geq \kappa \} 
   &\geq \mu \left(  \{ x \,:\, |h(x)| \geq \kappa^t \} \cap \left\{ x \,:\, 1 - \frac{2}{\kappa} \geq x \geq \frac{2}{\kappa} \right\}  \right)\\
  &\geq \mu \{ x \,:\, |h(x)| \geq \kappa^t \} - \mu \left( \left\{ x \,:\, 1 - \frac{2}{\kappa} \geq x \geq \frac{2}{\kappa} \right\}^c \right) \\
  &\geq  \mu \{ x \,:\, |h(x)| \geq \kappa^t \} - \frac{4}{\kappa}
\end{align*}

Hence, 
\begin{align*}
\mu \{ x \,:\, |h(x)| \geq \kappa^t \} \leq \frac{M'+4}{\kappa} 
\end{align*}

A change of variables gives the desired result.

\end{proof}
